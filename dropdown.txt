<!DOCTYPE html>
<html>
  <head>
    <meta charset="utf-8">
    <style>
      .dropbtn{
        border-radius: 10px;
        background-color: navy;
        color: white;
        font-size: 15px;
        width: 100px;
        height: 50px;
        cursor: pointer;
      }

      .dropdown{
        display: inline-block;
      }

      .dropdown-content{
        position: absolute;
        background-color: #dedede;
        border-radius: 10px;
        display: none;
        width: auto;
        box-shadow: 0px 8px 16px 0px rgba(0,0,0,0.2);
        z-index: 1;
      }

      .dropdown-content a{
        display: block;
        text-decoration: none;
        color: black;
        padding: 10px 10px;
      }

      .dropdown-content a:hover {
        background-color: #929fae;
        border-radius: 10px;
      }

      .btn_toggle{
        display: block;
      }

    </style>

    <script>
      // https://www.w3schools.com/howto/howto_js_dropdown.asp
      // class="dropdown-content" 에게 class="btn_toggle" 을 true/false 해주는 기능
      function btnDropdown(){
        document.querySelector('.dropdown-content').classList.toggle('btn_toggle');
      }

      window.addEventListener("click", function(event){
        if(!event.target.matches('.dropbtn')){
          var dropdowns = document.querySelector('.dropdown-content');
          var cList = dropdowns.classList;
          if(cList[1] == 'btn_toggle'){
            btnDropdown();
          }
        }
      });

    </script>

  </head>
  <body>
    <div class="dropdown">
      <input type="button" value="Dropdown" class="dropbtn" onclick="
        btnDropdown();
      ">
      <div class="dropdown-content">
        <a target="_blank" href="https://www.w3schools.com/howto/howto_js_dropdown.asp">Clickable Dropdown Example</a>
        <a href="#">LINK 2</a>
      </div>
    </div>
